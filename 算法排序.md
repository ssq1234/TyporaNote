```java
package recursion;

/**
 * 递归 迷宫回溯问题
 * 小球能找到map[6][5] 即找到通路
 * 1表示墙不可以走
 * 0表示该点没有走过
 * 2表示该点是通路
 * 3表示该位置已经走过但是走不通
 * 在走迷宫时 下->右->上->左 行走策略 如果该点走不通再回溯
 *
 * @author 孙术强
 * @date 2021/2/13 12:42
 */
public class MazeBack {
    public static void main(String[] args) {
        // 先创建一个二维数组模拟迷宫
        // 地图
        int[][] map = new int[8][7];

        // 先把上下全都变成1
        for (int i = 0; i < 7; i++) {
            map[0][i] = 1;
            map[7][i] = 1;
        }
        // 左右都变成1
        for (int i = 0; i < 8; i++) {
            map[i][0] = 1;
            map[i][6] = 1;
        }
        // 设置挡板
        map[3][1] = map[3][2] = map[1][2] = map[3][3]= 1;

        // 输出地图
        showMap(map);

        // 找通路
        setWay(map, 1, 1);

        // 输出地图
        showMap(map);
    }


    // 输出地图
    public static void showMap(int[][] map) {
        System.out.println("地图的情况");
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 7; j++) {
                System.out.print(map[i][j] + " ");
            }
            System.out.println();
        }
    }


    /**
     * 使用递归找路
     *
     * @param map 表示地图
     * @param i   表示从哪里开始找
     * @param j   表示从哪里开始找
     * @return 找到通路就返回true 没找到返回false
     */
    public static boolean setWay(int[][] map, int i, int j) {
        if (map[6][5] == 2) {   // 表示通路已经找到了
            return true;
        } else {
            if (map[i][j] == 0) { // 如果当前这个点还没有走过
                map[i][j] = 2; // 假设该点是可以走通的
                // 按照默认行走策略
                if (setWay(map, i + 1, j)) { // 向下走
                    return true;
                } else if (setWay(map, i, j + 1)) { // 向右走
                    return true;
                } else if (setWay(map, i - 1, j)) { // 向上走
                    return true;
                } else if (setWay(map, i, j - 1)) { // 向左走
                    return true;
                } else { // 说明该点是死路
                    map[i][j] = 3;
                    return false;
                }
            } else { // 如果map[i][j]!=0 可能1墙不能走，2表示已经走过了，3表示不通
                return false;
            }
        }
    }
}
```

```java
package eightqueen;

/**
 * 递归 回溯算法 八皇后问题
 * @author 孙术强
 * @date 2021/2/16 18:03
 */
public class EightQueen {

     // 定义一个max表示共有多少皇后
    int max = 8;
     // 定义数组arr，保存皇后放置位置的结果
    int[] arr = new int[max];
     // 一共有多少种解
    int count;
     // 冲突次数
    int unCount;
    public static void main(String[] args) {
        EightQueen eightQueen = new EightQueen();
        eightQueen.check(0);
        System.out.printf("一共有%d种解法", eightQueen.count);
        System.out.printf("一共有%d次冲突", eightQueen.unCount);
    }

    /**
     * 编写一个方法放置第n个皇后
     *
     */
    private void check(int n){
        if(n == max){ // n=8 所有的皇后都放好了
            print();
            count++;
            return;
        }
         // 依次放入皇后
        for (int i = 0; i < max; i++) {
             // 先把当前这个皇后 n 放到该行的第一列
            arr[n] =i;
             // 当防止第n个皇后到i列时是否冲突
            if (judge(n)){ // 不冲突
                // 接着放n+1个皇后
                check(n+1);
            }
             // 如果冲突继续执行 往后移一列
        }
    }

    /**
     * 查看当我们放置第n个皇后，就去监测该皇后是否和前面已经摆放的皇后冲突
     * @param n n表示第n个皇后
     * @return
     */
    private boolean judge(int n){
        for (int i = 0; i < n; i++) {
             // 是否在同一列或者同一斜线
            if (arr[i] == arr[n] || Math.abs(n-i) == Math.abs(arr[n]-arr[i])){
                unCount++;
                return false;
            }
        }
        return true;
    }

     // 将皇后摆放的位置输出
    private void print(){
        for (int i : arr) {
            System.out.print(i+" ");
        }
        System.out.println();
    }
}

```

### 1.冒泡排序

```java
package sort.bubbling;
import java.util.Arrays;
/**
 * 冒泡排序 时间复杂度 O(n^2)
 * 一共进行length-1轮排序
 * 第一轮进行length-1次比较 从第二轮开始比较次数逐渐-1
 * 如果后面的值小于前面的值 就进行交换
 * 最后得到一个升序排序结果
 * 如果需要比较的数据比较多 那么速度会非常缓慢
 * @author 孙术强
 * @date 2021/2/21 15:39
 */
public class BubblingSort {
    public static void main(String[] args) {
        int[] arr = {1,-1,2,-2,3,40}; // 需要排序的数组
        sort(arr);
    }
    private static void sort(int[] arr){
         // 最多进行length-1轮排序
        for (int i = 0; i < arr.length-1; i++) {
             // 默认本轮没有进行排序
            boolean judge = false;
             // 本轮需要比较length-1-i次
            for (int j = 0; j < arr.length-1-i; j++) {
                if (arr[j]>arr[j+1]){
                    judge = true;  // 标识本轮已经进行排序
                     // 换位操作
                    int temp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = temp;
                }
            }
            if (!judge){ // 如果本轮没有进行换位 说明已经排序完成 跳出循环
                System.out.println("排序中途结束！");
                break;
            }
            System.out.println("第"+(i+1)+"次排序结果");
            System.out.println(Arrays.toString(arr));
        }
    }
}
```

### 2.选择排序

```java
package sort.selection;

import java.util.Arrays;

/**
 * 选择排序 时间复杂度 O(n^2)
 * {101,34,119,1} 一共进行length-1轮
 * 第一轮 从一到四个当中选择一个最小的放在arr[0]
 * 第二轮 从二到四个当中选择一个最小的放在arr[1]
 * 第三轮 从三到四个当中选择一个最小的放在arr[2]
 * @author 孙术强
 * @date 2021/2/23 16:12
 */
public class SelectionSort {
    public static void main(String[] args) {
        int[] arr = {101,34,119,1};
        selectionSort(arr);
        System.out.println(Arrays.toString(arr));
    }


    private static void selectionSort(int[] arr){
        for (int i = 0; i < arr.length-1; i++) {
            int minIndex = i;
            for (int j = i+1; j < arr.length; j++) {
                if (arr[minIndex] > arr[j])
                    minIndex = j;
            }
            int temp = arr[minIndex];
            arr[minIndex] = arr[i];
            arr[i] = temp;
        }
    }
}
```
