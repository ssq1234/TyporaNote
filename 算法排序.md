package recursion;

/**
 * 递归 迷宫回溯问题
 * 小球能找到map[6][5] 即找到通路
 * 1表示墙不可以走
 * 0表示该点没有走过
 * 2表示该点是通路
 * 3表示该位置已经走过但是走不通
 * 在走迷宫时 下->右->上->左 行走策略 如果该点走不通再回溯
 *
 * @author 孙术强
 * @date 2021/2/13 12:42
 */
public class MazeBack {
    public static void main(String[] args) {
        // 先创建一个二维数组模拟迷宫
        // 地图
        int[][] map = new int[8][7];

        // 先把上下全都变成1
        for (int i = 0; i < 7; i++) {
            map[0][i] = 1;
            map[7][i] = 1;
        }
        // 左右都变成1
        for (int i = 0; i < 8; i++) {
            map[i][0] = 1;
            map[i][6] = 1;
        }
        // 设置挡板
        map[3][1] = map[3][2] = map[1][2] = map[3][3]= 1;

        // 输出地图
        showMap(map);

        // 找通路
        setWay(map, 1, 1);

        // 输出地图
        showMap(map);
    }


    // 输出地图
    public static void showMap(int[][] map) {
        System.out.println("地图的情况");
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 7; j++) {
                System.out.print(map[i][j] + " ");
            }
            System.out.println();
        }
    }


    /**
     * 使用递归找路
     *
     * @param map 表示地图
     * @param i   表示从哪里开始找
     * @param j   表示从哪里开始找
     * @return 找到通路就返回true 没找到返回false
     */
    public static boolean setWay(int[][] map, int i, int j) {
        if (map[6][5] == 2) {   // 表示通路已经找到了
            return true;
        } else {
            if (map[i][j] == 0) { // 如果当前这个点还没有走过
                map[i][j] = 2; // 假设该点是可以走通的
                // 按照默认行走策略
                if (setWay(map, i + 1, j)) { // 向下走
                    return true;
                } else if (setWay(map, i, j + 1)) { // 向右走
                    return true;
                } else if (setWay(map, i - 1, j)) { // 向上走
                    return true;
                } else if (setWay(map, i, j - 1)) { // 向左走
                    return true;
                } else { // 说明该点是死路
                    map[i][j] = 3;
                    return false;
                }
            } else { // 如果map[i][j]!=0 可能1墙不能走，2表示已经走过了，3表示不通
                return false;
            }
        }
    }
}
